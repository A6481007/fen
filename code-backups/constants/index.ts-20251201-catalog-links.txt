export type NavKind = "primary" | "download" | "event" | "resource";

type NavMetadata = {
  badge?: string;
  external?: boolean;
  icon?: string;
};

export type NavChild = NavMetadata & {
  label: string;
  path: string;
  description?: string;
  kind?: NavKind;
};

export type NavItem = NavMetadata & {
  label: string;
  path: string;
  children?: NavChild[];
};

export type HeaderNavChild = {
  title: string;
  href: string;
  description?: string;
  kind?: NavKind;
  badge?: string;
  external?: boolean;
};

export type HeaderNavItem = {
  title: string;
  href: string;
  children?: HeaderNavChild[];
  badge?: string;
  external?: boolean;
};
